       // Compute baseline angle in screen space (CW from right) per RESEARCH.md §Intersection Math
       // Use mapper to convert line endpoints to screen, then Atan2
       var p1Screen = mapper.PageToScreen(AnchorLine.X1Pt, AnchorLine.Y1Pt);
       var p2Screen = mapper.PageToScreen(AnchorLine.X2Pt, AnchorLine.Y2Pt);
       double screenDx = p2Screen.X - p1Screen.X;

       double baselineAngleDeg = Math.Atan2(screenDy, screenDx) * 180.0 / Math.PI;

       var protractor = new ProtractorObject(

           baselineAngleDeg,
           AnchorLine.Id, line2.Id);

       _geometryService.ExecuteCommand(new PlaceObjectCommand(protractor));
       _geometryService.SetSelected(protractor.Id);

       break;

Add the new private method `DrawGhostProtractor`:

```csharp
/// <summary>
/// Draws a semi-transparent protractor arc at the cursor position during placement (click 2).
/// This gives the student a preview of what will be placed.
/// Uses transient SKPaint objects (via using) — this is acceptable for a ghost/transient draw
/// since this code runs only during mouse moves, not per-frame with geometry already placed.
/// </summary>
private void DrawGhostProtractor(SKCanvas canvas, SKPoint centerPx)
{
    if (_coordinateMapper is null) return;

    // Compute display radius in screen pixels from DefaultRadiusPt + current scale
    float radiusPx = (float)(ProtractorObject.DefaultRadiusPt * _coordinateMapper.Scale);
    if (radiusPx < 20f) return; // too small to be useful

    using var ghostArcPaint = new SKPaint
    {
        Style       = SKPaintStyle.Stroke,
        Color       = new SKColor(0x3B, 0x6F, 0xD4, 128), // cobalt at 50% alpha
        StrokeWidth = 2f,
        IsAntialias = true,
    };

    // Draw a simple semicircle arc to represent the protractor ghost
    // startAngle -180° = 9 o'clock (left end), sweepAngle 180° = top semicircle
    var oval = new SKRect(
        centerPx.X - radiusPx, centerPx.Y - radiusPx,
        centerPx.X + radiusPx, centerPx.Y + radiusPx);
    canvas.DrawArc(oval, -180f, 180f, false, ghostArcPaint);

    // Baseline
    canvas.DrawLine(
        centerPx.X - radiusPx, centerPx.Y,
        centerPx.X + radiusPx, centerPx.Y,
        ghostArcPaint);

    // Center dot
    using var ghostDotPaint = new SKPaint
    {
        Style       = SKPaintStyle.Fill,
        Color       = new SKColor(0x3B, 0x6F, 0xD4, 128),
        IsAntialias = true,
    };
    canvas.DrawCircle(centerPx, 4f, ghostDotPaint);
}
```
