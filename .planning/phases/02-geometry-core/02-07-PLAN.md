**Style 1 — RailButtonStyle** (for nudge pad directional buttons, delete, undo, redo):
- Gaze target: explicit Width/Height set per button in XAML (56x56 for nudge/undo/redo)
- Appearance: white surface, BrushBorder outline, CornerRadius=6, BrushInk text, FontBody
- Hover: BrushSurface2 background
- Pressed: BrushAccentSoft background + BrushAccent border
- Disabled: 40% opacity

```xml
<!-- ── RailButton style: right rail action buttons (nudge pad, delete, undo/redo) ── -->
<!-- Gaze-safe chrome: white surface, subtle border, no WPF default gray chrome. -->
<Style x:Key="RailButtonStyle" TargetType="Button">
    <Setter Property="Background"      Value="{StaticResource BrushSurface}"/>
    <Setter Property="BorderBrush"     Value="{StaticResource BrushBorder}"/>
    <Setter Property="BorderThickness" Value="1.5"/>
    <Setter Property="Foreground"      Value="{StaticResource BrushInk}"/>
    <Setter Property="FontFamily"      Value="{StaticResource FontBody}"/>
    <Setter Property="FontSize"        Value="13"/>
    <Setter Property="Cursor"          Value="Hand"/>
    <Setter Property="FocusVisualStyle" Value="{x:Null}"/>
    <Setter Property="Template">
        <Setter.Value>
            <ControlTemplate TargetType="Button">
                <Border x:Name="Bd"
                        Background="{TemplateBinding Background}"
                        BorderBrush="{TemplateBinding BorderBrush}"
                        BorderThickness="{TemplateBinding BorderThickness}"
                        CornerRadius="6">
                    <ContentPresenter HorizontalAlignment="Center" VerticalAlignment="Center"/>
                </Border>
                <ControlTemplate.Triggers>
                    <Trigger Property="IsMouseOver" Value="True">
                        <Setter TargetName="Bd" Property="Background" Value="{StaticResource BrushSurface2}"/>
                    </Trigger>
                    <Trigger Property="IsPressed" Value="True">
                        <Setter TargetName="Bd" Property="Background" Value="{StaticResource BrushAccentSoft}"/>
                        <Setter TargetName="Bd" Property="BorderBrush" Value="{StaticResource BrushAccent}"/>
                    </Trigger>
                    <Trigger Property="IsEnabled" Value="False">
                        <Setter Property="Opacity" Value="0.4"/>
                    </Trigger>
                </ControlTemplate.Triggers>
            </ControlTemplate>
        </Setter.Value>
    </Setter>
</Style>
```

**Style 2 — StepButtonStyle** (for 1px / 5px / 20px step selector buttons):
- Base: same as RailButtonStyle (white, BrushBorder, CornerRadius=6)
- Active state: set via Tag property — `Tag="active"` → BrushAccent background + white text
- Do NOT use a DataTrigger here (styles cannot bind to ViewModel directly); the active state Tag
  will be set in RightRail.xaml using DataTriggers on the Button element itself

```xml
<!-- ── StepButton style: nudge step selector (1px / 5px / 20px) ── -->
<!-- Base appearance matches RailButtonStyle. Active state (Tag="active") -->
<!-- uses accent cobalt background + white text. Tag is set from RightRail.xaml DataTriggers. -->
<Style x:Key="StepButtonStyle" TargetType="Button">
    <Setter Property="Background"      Value="{StaticResource BrushSurface}"/>
    <Setter Property="BorderBrush"     Value="{StaticResource BrushBorder}"/>
    <Setter Property="BorderThickness" Value="1.5"/>
    <Setter Property="Foreground"      Value="{StaticResource BrushInk}"/>
    <Setter Property="FontFamily"      Value="{StaticResource FontBody}"/>
    <Setter Property="FontSize"        Value="12"/>
    <Setter Property="FontWeight"      Value="SemiBold"/>
    <Setter Property="Cursor"          Value="Hand"/>
    <Setter Property="FocusVisualStyle" Value="{x:Null}"/>
    <Setter Property="Template">
        <Setter.Value>
            <ControlTemplate TargetType="Button">
                <Border x:Name="Bd"
                        Background="{TemplateBinding Background}"
                        BorderBrush="{TemplateBinding BorderBrush}"
                        BorderThickness="{TemplateBinding BorderThickness}"
                        CornerRadius="6">
                    <ContentPresenter HorizontalAlignment="Center" VerticalAlignment="Center"/>
                </Border>
                <ControlTemplate.Triggers>
                    <Trigger Property="Tag" Value="active">
                        <Setter TargetName="Bd" Property="Background" Value="{StaticResource BrushAccent}"/>
                        <Setter TargetName="Bd" Property="BorderBrush" Value="{StaticResource BrushAccent}"/>
                        <Setter Property="Foreground" Value="White"/>
                    </Trigger>
                    <MultiTrigger>
                        <MultiTrigger.Conditions>
                            <Condition Property="IsMouseOver" Value="True"/>
                            <Condition Property="Tag" Value="active"/>
                        </MultiTrigger.Conditions>
                        <Setter TargetName="Bd" Property="Background" Value="{StaticResource BrushAccentInk}"/>
                    </MultiTrigger>
                    <MultiTrigger>
                        <MultiTrigger.Conditions>
                            <Condition Property="IsMouseOver" Value="True"/>
                        </MultiTrigger.Conditions>
                        <MultiTrigger.Setters>
                            <Setter TargetName="Bd" Property="Background" Value="{StaticResource BrushSurface2}"/>
                        </MultiTrigger.Setters>
                    </MultiTrigger>
                    <Trigger Property="IsPressed" Value="True">
                        <Setter TargetName="Bd" Property="Background" Value="{StaticResource BrushAccentSoft}"/>
                    </Trigger>
                </ControlTemplate.Triggers>
            </ControlTemplate>
        </Setter.Value>
    </Setter>
</Style>
```

```xml
<UniformGrid Rows="1" Columns="3" Margin="0,0,0,6">
```

```xml
            Style="{StaticResource StepButtonStyle}"
            Command="{Binding SetStepCommand}" CommandParameter="1"
            Content="1px">
        <Button.Style>
            <Style TargetType="Button" BasedOn="{StaticResource StepButtonStyle}">
                <Style.Triggers>
                    <DataTrigger Binding="{Binding NudgeStepPx}" Value="1">
                        <Setter Property="Tag" Value="active"/>
                    </DataTrigger>
                </Style.Triggers>
            </Style>
        </Button.Style>
    </Button>
    <Button Height="40"
            Style="{StaticResource StepButtonStyle}"
            Command="{Binding SetStepCommand}" CommandParameter="5"
            Content="5px">
        <Button.Style>
            <Style TargetType="Button" BasedOn="{StaticResource StepButtonStyle}">
                <Style.Triggers>
                    <DataTrigger Binding="{Binding NudgeStepPx}" Value="5">
                        <Setter Property="Tag" Value="active"/>
                    </DataTrigger>
                </Style.Triggers>
            </Style>
        </Button.Style>
    </Button>
    <Button Height="40"
            Style="{StaticResource StepButtonStyle}"
            Command="{Binding SetStepCommand}" CommandParameter="20"
            Content="20px">

</UniformGrid>
```

IMPORTANT: When using `Button.Style` inline with `BasedOn`, do NOT also set `Style="{StaticResource StepButtonStyle}"` on the outer Button element — the inline style already inherits from StepButtonStyle via BasedOn.
